## Display Joint Positions

Create `Assets/Scripts/JointUI.cs`:

```csharp
using Unity.Robotics.ROSTCPConnector;
using RosMessageTypes.Sensor;
using UnityEngine;
using UnityEngine.UI;

public class JointUI : MonoBehaviour
{
    public Text jointDataText;

    void Start()
    {
        ROSConnection.GetOrCreateInstance().Subscribe<JointStateMsg>("/joint_states", UpdateUI);
    }

    void UpdateUI(JointStateMsg msg)
    {
        string display = "Joint Positions:\n";
        for (int i = 0; i < msg.name.Length; i++)
        {
            display += $"{msg.name[i]}: {msg.position[i]:F2}\n";
        }
        jointDataText.text = display;
    }
}
```

Add UI Text:

2. Position: Top-left corner
3. Assign to script in Inspector

## Connection Status Indicator

```csharp
using UnityEngine;
using UnityEngine.UI;

public class ConnectionStatus : MonoBehaviour
{
    public Image statusIndicator;
    float lastMessageTime;

    void UpdateStatus()
    {
        if (Time.time - lastMessageTime < 1f)
        {
            statusIndicator.color = Color.green; // Connected
        }
        else
        {
            statusIndicator.color = Color.red; // Disconnected
        }
    }
}
```

## Control Buttons

Send commands to Gazebo:

```csharp
using Unity.Robotics.ROSTCPConnector;
using RosMessageTypes.Std;
using UnityEngine;
using UnityEngine.UI;

public class ResetButton : MonoBehaviour
{
    void Start()
    {
        GetComponent<Button>().onClick.AddListener(OnResetClicked);
    }

    void OnResetClicked()
    {
        ROSConnection.GetOrCreateInstance().Publish("/reset_simulation", new EmptyMsg());
        Debug.Log("Reset command sent");
    }
}
```

## Performance Metrics

Display FPS and latency:

```csharp
public class PerformanceUI : MonoBehaviour
{
    public Text metricsText;

    void Update()
    {
        fps = 1f / Time.deltaTime;
```
